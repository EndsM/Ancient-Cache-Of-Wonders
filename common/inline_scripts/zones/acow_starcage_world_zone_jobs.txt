# We don't modify the district based jobs adding, instead, create jobs from mod's zones
# fallback
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_gestalt = no
		}
	}
	# Unlink AOW, we can step base value based on district core/secondary and level
	job_acot_enforcer_add = $number$
	mult = value:AOW_var_starcage_world_points|v|AOW_var_sw_points|
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_gestalt = yes
		}
	}

	mult = value:AOW_var_starcage_world_points|v|AOW_var_sw_points|
}
