acow_btn_sw_reset_all = {
	potential = {
		from = {
			has_active_event = {
				acow_starcage_world_events.1
			}
		}
	}
	allow = {
		always = yes
	}
	effect = {
		hidden_effect = {
			# Reset all ACOT specific job variables to 0
			set_variable = {
				which = acow_var_sw_points_acot_dark_energy
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_dark_matter
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_researcher
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_energy
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_mineral
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_food
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_transmute
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_alloy
				value = 0
			}
			set_variable = {
				which = acow_var_sw_points_acot_artisan
				value = 0
			}
			# Restore the ACOW unallocated pool to max
			set_variable = {
				which = acow_var_sw_max_points
				value = @acow_var_max_sw_points_base
			}
		}
	}
}
