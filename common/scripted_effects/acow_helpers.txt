acow_init_if_not = {
	if = {
		limit = {
			NOT = {
				is_variable_set = $key$
			}
		}
		set_variable = {
			which = $key$
			value = $value$
		}
	}
}

acow_check_do_we_need_update = {
	if = {
		limit = {
			is_variable_set = acow_var_sw_max_points_log
			is_variable_set = acow_var_sw_max_points
			value:acow_var_max_sw_points_dynamic > acow_var_sw_max_points_log
		}
		set_variable = {
			which = acow_temp_diff
			value = value:acow_var_max_sw_points_dynamic
		}
		subtract_variable = {
			which = acow_temp_diff
			value = acow_var_sw_max_points_log
		}
		change_variable = {
			which = acow_var_sw_max_points
			value = acow_temp_diff
		}
		set_variable = {
			which = acow_var_sw_max_points_log
			value = value:acow_var_max_sw_points_dynamic
		}
		# Clean the temp
		clear_variable = acow_temp_diff
	}
}
